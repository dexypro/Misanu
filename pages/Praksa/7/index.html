Приказан поступак креирања графа и имплементација основних метода графа

Grafovi


package grafovi;

import java.util.LinkedList;

public class Graf {
    // Broj čvorova u grafu
    private int brojCvorova;

    // Lista susedstva za čuvanje veza između čvorova
    private LinkedList[] listaSusedstva;

    public Graf(int brojCvorova) {
        this.brojCvorova = brojCvorova;
        listaSusedstva = new LinkedList[brojCvorova];

        for (int i = 0; i < brojCvorova; i++) {
            listaSusedstva[i] = new LinkedList<>();
        }
    }

    // Dodavanje grane između dva čvora u grafu
    public void dodajGranu(int u, int v) {
        listaSusedstva[u].add(v);
        listaSusedstva[v].add(u);
    }

    // Vraća broj čvorova u grafu
    public int brojCvorova() {
        return brojCvorova;
    }

    // Vraća broj grana u grafu
    public int brojGrana() {
        int brojGrana = 0;
        for (int i = 0; i < brojCvorova; i++) {
            brojGrana += listaSusedstva[i].size();
        }
        // Delimo sa 2, jer je svaka grana prebrojana dva puta (jednom za svaki čvor)
        return brojGrana / 2;
    }

    // Vraća listu susednih čvorova za zadati čvor
    public LinkedList susedniCvorovi(int cvor) {
        return listaSusedstva[cvor];
    }

    public static void main(String[] args) {
        // Kreiramo graf sa 5 čvorova
        Graf graf = new Graf(5);

        // Dodajemo grane između čvorova
        graf.dodajGranu(0, 1);
        graf.dodajGranu(0, 4);
        graf.dodajGranu(1, 3);
        graf.dodajGranu(1, 4);
        graf.dodajGranu(2, 3);
        graf.dodajGranu(3, 4);

        System.out.println("Broj čvorova: " + graf.brojCvorova());
        System.out.println("Broj grana: " + graf.brojGrana());

        for (int i = 0; i < graf.brojCvorova(); i++) {
            LinkedList susedi = graf.susedniCvorovi(i);
            System.out.print("Susedi čvora " + i + ":");
            for (Integer sused : susedi) {
                System.out.print(" " + sused);
            }
            System.out.println();
        }
    }

}
